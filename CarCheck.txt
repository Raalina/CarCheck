      RebateTime@1004 : Decimal;
      Customer@1003 : Record 18;
      DefaultTime@1002 : Decimal;
      UnregDuration@1001 : Decimal;
      PaymentFound@1000 : Boolean;
    BEGIN
      RebateTime := 0.25;
      DefaultTime := 12;
      PaymentFound := TRUE;

      IF (TimeDate <> 0DT) AND (Place <> '') THEN BEGIN
        Customer.SETRANGE(Name, "Driver Name");
        "Posted Route Payment".SETRANGE("Driver Name", Customer.Name);

        IF NOT "Posted Route Payment".FINDLAST THEN BEGIN
          Time := DefaultTime;
          PaymentFound := FALSE

        IF Time > DefaultTime THEN BEGIN
          Time := DefaultTime;
          PaymentFound := FALSE;

            Sum := Sum / 2;
        END;
      END;
    END;
